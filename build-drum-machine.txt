** start of script.js **

function playSound(id) {
  const audio = document.getElementById(id);
  if (audio) {
    audio.currentTime = 0;
    audio.play();
    document.getElementById("display").innerText = audio.parentElement.id;
  }
}

document.onkeydown = function(e) {
  const char = e.key.toUpperCase();
  playSound(char);
};
